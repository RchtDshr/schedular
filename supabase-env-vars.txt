# Environment Variables for Supabase Edge Function
# Copy these to your Supabase Dashboard → Settings → Environment Variables

# Email Service (Resend)
RESEND_API_KEY=re_your_resend_api_key_here
FROM_EMAIL=noreply@yourdomain.com

# Security
CRON_SECRET=your_secure_random_string_here

# Your deployed Next.js application URL
NEXTJS_API_URL=https://yourdomain.com

# MongoDB connection string
MONGODB_URI=mongodb+srv://username:password@cluster.mongodb.net/database

# Notes:
# 1. Replace all placeholder values with your actual values
# 2. CRON_SECRET should be a secure random string (use a password generator)
# 3. NEXTJS_API_URL should point to your deployed Next.js app
# 4. Get RESEND_API_KEY from https://resend.com/api-keys
